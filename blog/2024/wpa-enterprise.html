1. 修改配置文件

vim /etc/NetworkManager/system-connections/你的wifi名字.nmconnection
在 [802-1x]
新增 phase1-auth-flags=32

2. 配置文件

2.1. 文件路径

/etc/NetworkManager/system-connections/你的wifi名字.nmconnection

2.2. example

[connection]
id=xxxxxxx
uuid=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx
type=wifi
interface-name=wlp0s20f3

[wifi]
mode=infrastructure
ssid=xxxxxx

[wifi-security]
auth-alg=open
key-mgmt=wpa-eap

[802-1x]
eap=peap;
identity=username
password=userpwd
phase1-auth-flags=32
phase2-auth=mschapv2

[ipv4]
method=auto

[ipv6]
addr-gen-mode=default
method=auto

[proxy]
